Access all BlackRoad services through our unified API. One API key, 153+ products, 500+ endpoints. 

// Initialize the BlackRoad SDK
import { BlackRoad } from '@blackroad/sdk';

const client = new BlackRoad({
  apiKey: 'sk_live_your_key_here',
  endpoint: 'https://api.blackroad.io/v1'
});

// Authenticate a user
const session = await client.auth.signIn({
  email: 'user@example.com',
  password: 'secure_password'
});

// Generate AI response
const response = await client.ai.generate({
  model: 'claude-sonnet-4',
  prompt: 'Explain quantum computing',
  temperature: 0.7
});
